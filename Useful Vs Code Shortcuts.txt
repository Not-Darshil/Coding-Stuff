QUICK FORMATTING = SHIFT + ALT + F
WORD WRAP = ALT + Z
OPEN FOLDER = CTRL + K + O
OPEN FILE = CTRL + O
COPY AND PASTE A LINE = LEFT ALT + LEFT SHIFT + DOWN KEY
HOLD ALT FOR MULTIPLE CURSORS = ALT
MOVE THE LINES = LEFT ALT + ARROW
CHANGE KEYBOARD SHORTCUTS = CTRL + K + S
CHANGE TEXT FILES = ALT + LEFT/RIGHT
FIND AND REPLACE = CTRL + F + H
